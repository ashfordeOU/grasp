```typescript
export type OrmType = 'prisma' | 'typeorm' | 'sequelize' | 'sqlalchemy';

export interface OrmQuery {
  file: string;
  line: number;
  orm: OrmType;
  model: string;
  operation: string;
  confidence: number;
}

interface FileSnippet {
  path: string;
  content: string | null;
  isCode: boolean;
}

export function detectOrmQueries(files: FileSnippet[]): OrmQuery[] {
  const results: OrmQuery[] = [];

  for (const file of files) {
    if (!file.isCode || !file.content) continue;
    const lines = file.content.split('\n');

    // Prisma: prisma.model.operation(...)
    const prismaRe = /\bprisma\.(\w+)\.(findMany|findFirst|findUnique|findUniqueOrThrow|create|createMany|update|updateMany|upsert|delete|deleteMany|count|aggregate|groupBy)\b/g;
    for (let i = 0; i < lines.length; i++) {
      prismaRe.lastIndex = 0;
      let m: RegExpExecArray | null;
      while ((m = prismaRe.exec(lines[i])) !== null) {
        results.push({ file: file.path, line: i + 1, orm: 'prisma', model: m[1], operation: m[2], confidence: 0.95 });
      }
    }

    // TypeORM: repository.find/save/etc or getRepository(Entity).find
    const typeormRe = /\b(?:this\.)?(?:(?:\w+Repository|\w+Repo|repository|getRepository\(\w+\))\.(find|findOne|findOneBy|findBy|findAndCount|save|insert|update|delete|remove|softDelete|restore|count|exist|query))\b/g;
    for (let i = 0; i < lines.length; i++) {
      typeormRe.lastIndex = 0;
      let m: RegExpExecArray | null;
      while ((m = typeormRe.exec(lines[i])) !== null) {
        results.push({ file: file.path, line: i + 1, orm: 'typeorm', model: 'unknown', operation: m[1], confidence: 0.80 });
      }
    }

    // Sequelize: Model.findAll / Model.create etc
    const seqRe = /\b(\w+)\.(findAll|findOne|findByPk|findOrCreate|create|bulkCreate|update|destroy|count|max|min|sum)\s*\(/g;
    for (let i = 0; i < lines.length; i++) {
      seqRe.lastIndex = 0;
      let m: RegExpExecArray | null;
      while ((m = seqRe.exec(lines[i])) !== null) {
        // Skip common false positives (array methods, string methods)
        const name = m[1];
        if (['arr', 'array', 'str', 'string', 'obj', 'res', 'req', 'items', 'data', 'list'].includes(name.toLowerCase())) continue;
        results.push({ file: file.path, line: i + 1, orm: 'sequelize', model: name, operation: m[2], confidence: 0.70 });
      }
    }

    // SQLAlchemy: session.query / db.execute / session.add
    const saRe = /\b(?:session|db|Session|conn)\.(query|execute|add|delete|merge|bulk_insert_mappings|bulk_update_mappings|bulk_save_objects|commit|rollback)\s*\(/g;
    for (let i = 0; i < lines.length; i++) {
      saRe.lastIndex = 0;
      let m: RegExpExecArray | null;
      while ((m = saRe.exec(lines[i])) !== null) {
        results.push({ file: file.path, line: i + 1, orm: 'sqlalchemy', model: 'unknown', operation: m[1], confidence: 0.75 });
      }
    }
  }

  return results;
}
```

```typescript
async indexOrmQueries(
  repoId: string,
  queries: import('./orm-tracker.js').OrmQuery[],
  fnByNameAndFile: Map<string, string>,
): Promise<void> {
  await this.ready;
  for (const q of queries) {
    const fileNodeId = `${repoId}:${q.file}`;
    // Find any function in this file to attach the edge to (use first fn in file, or file itself)
    const fnId = fnByNameAndFile.get(`${q.file}::__file_query__`) ??
      [...fnByNameAndFile.entries()].find(([k]) => k.startsWith(q.file + '::'))?.[1];
    if (!fnId) continue;
    try {
      const res = await this.conn.query(
        `MATCH (f:Function {id: '${esc(fnId)}'}), (fp:File {id: '${esc(fileNodeId)}'}) CREATE (f)-[:QUERIES {orm: '${esc(q.orm)}', model: '${esc(q.model)}', operation: '${esc(q.operation)}'}]->(fp)`
      );
      await res.close();
    } catch { /* skip if nodes missing */ }
  }
}
```

```typescript
// =====================================================================
// TOOL: grasp_orm_map
// =====================================================================
server.registerTool(
  'grasp_orm_map',
  {
    title: 'ORM Query Map',
    description: `Detects ORM query patterns across all source files. Supports Prisma, TypeORM, Sequelize, and SQLAlchemy. Groups results by ORM framework and model name. Use to understand data access patterns, find N+1 risks, or map which models are queried where.`,
    inputSchema: z.object({
      session_id: z.string().describe('Session ID from grasp_analyze'),
      orm_filter: z.enum(['prisma', 'typeorm', 'sequelize', 'sqlalchemy', 'all']).default('all'),
    }).strict(),
    annotations: { readOnlyHint: true, destructiveHint: false, idempotentHint: true, openWorldHint: false },
  },
  async ({ session_id, orm_filter }) => {
    const data = await getSession(session_id);
    if (!data) return { content: [{ type: 'text', text: `Session ${session_id} not found.` }] };

    const allQueries = detectOrmQueries(data.files);
    const queries = orm_filter === 'all' ? allQueries : allQueries.filter(q => q.orm === orm_filter);

    // Group by ORM
    const byOrm: Record<string, number> = {};
    for (const q of queries) byOrm[q.orm] = (byOrm[q.orm] ?? 0) + 1;

    // Group by model
    const byModel: Record<string, { count: number; operations: string[]; files: string[] }> = {};
    for (const q of queries) {
      if (!byModel[q.model]) byModel[q.model] = { count: 0, operations: [], files: [] };
      byModel[q.model].count++;
      if (!byModel[q.model].operations.includes(q.operation)) byModel[q.model].operations.push(q.operation);
      if (!byModel[q.model].files.includes(q.file)) byModel[q.model].files.push(q.file);
    }

    // Top hotspot files
    const byFile: Record<string, number> = {};
    for (const q of queries) byFile[q.file] = (byFile[q.file] ?? 0) + 1;
    const hotFiles = Object.entries(byFile).sort((a, b) => b[1] - a[1]).slice(0, 10).map(([f, c]) => ({ file: f, count: c }));

    const out = {
      total_queries: queries.length,
      by_orm: byOrm,
      by_model: byModel,
      hot_files: hotFiles,
      samples: queries.slice(0, 30),
    };
    return { content: [{ type: 'text', text: JSON.stringify(out) }], structuredContent: out };
  }
);
```

```typescript
import type { AnalysisResult, AnalyzedFile } from './types.js';
import { detectOrmQueries, type OrmQuery } from './orm-tracker.js';
import { propagateTypes, type PropagatedType } from './type-propagator.js';
import { buildScopeIndex, type ScopeIndex } from './scope-resolver.js';

export interface PipelineEnrichment {
  ormQueries: OrmQuery[];
  propagatedTypes: PropagatedType[];
  scopeIndex: ScopeIndex;
  phaseTimings: Record<string, number>;
}

export interface PipelinePhase {
  name: string;
  run(result: AnalysisResult, enrichment: Partial<PipelineEnrichment>): Promise<Partial<PipelineEnrichment>>;
}

const PHASES: PipelinePhase[] = [
  {
    name: 'scope',
    async run(result, enrichment) {
      const fnIds = new Map<string, string>();
      for (const f of result.files) {
        for (const fn of f.functions) {
          fnIds.set(`${f.path}::${fn.name}`, `${f.path}:${fn.name}:${fn.line}`);
        }
      }
      return { scopeIndex: buildScopeIndex(result.files, fnIds) };
    },
  },
  {
    name: 'type-propagation',
    async run(result) {
      return { propagatedTypes: propagateTypes(result.files, result.connections) };
    },
  },
  {
    name: 'orm',
    async run(result) {
      return { ormQueries: detectOrmQueries(result.files) };
    },
  },
];

export async function enrichAnalysis(result: AnalysisResult): Promise<PipelineEnrichment> {
  const enrichment: Partial<PipelineEnrichment> = { phaseTimings: {} };

  for (const phase of PHASES) {
    const t0 = Date.now();
    try {
      const partial = await phase.run(result, enrichment);
      Object.assign(enrichment, partial);
    } catch (e) {
      process.stderr.write(`[pipeline] phase ${phase.name} failed: ${(e as Error).message}\n`);
    }
    enrichment.phaseTimings![phase.name] = Date.now() - t0;
  }

  return enrichment as PipelineEnrichment;
}

export { PHASES };
```

```typescript
// =====================================================================
// TOOL: grasp_detect_changes
// =====================================================================
server.registerTool(
  'grasp_detect_changes',
  {
    title: 'Detect Changed Symbols',
    description: `Maps git diff output to affected symbols and execution processes. Given a local repo path and scope, runs git diff, extracts changed line ranges, and cross-references them with the indexed function table in the Grasp brain store.

scope options:
- unstaged: git diff (working tree vs index)
- staged: git diff --cached (index vs HEAD)
- all: git diff HEAD (all uncommitted changes)
- compare: git diff {base_ref}...HEAD (branch comparison)

Returns: changed files, affected symbols (with file+line), affected processes (BFS-traced from entry points), and a risk level (LOW/MEDIUM/HIGH/CRITICAL).`,
    inputSchema: z.object({
      source: z.string().describe('Local repo path (e.g. "/path/to/repo"). Remote repos not supported.'),
      scope: z.enum(['unstaged', 'staged', 'all', 'compare']).default('all'),
      base_ref: z.string().optional().describe('Base ref for compare scope (e.g. "main", "v1.0.0")'),
    }).strict(),
    annotations: { readOnlyHint: true, destructiveHint: false, idempotentHint: true, openWorldHint: true },
  },
  async ({ source, scope, base_ref }) => {
    // Only works for local paths
    const repoPath = source.startsWith('/') || source.startsWith('.') ? path.resolve(source) : null;
    if (!repoPath) {
      return { content: [{ type: 'text', text: 'grasp_detect_changes requires a local repo path, not a GitHub URL.' }] };
    }

    // Run git diff
    let diffOutput: string;
    try {
      const gitCmd = scope === 'unstaged' ? 'git diff' :
        scope === 'staged' ? 'git diff --cached' :
        scope === 'compare' ? `git diff ${base_ref ?? 'HEAD~1'}...HEAD` :
        'git diff HEAD';
      diffOutput = execSync(`${gitCmd} --unified=0`, { cwd: repoPath, maxBuffer: 10 * 1024 * 1024 }).toString();
    } catch (e: any) {
      if (e.message?.includes('not a git repository')) {
        return { content: [{ type: 'text', text: `Not a git repository: ${repoPath}` }] };
      }
      return { content: [{ type: 'text', text: `git diff failed: ${(e as Error).message}` }] };
    }

    if (!diffOutput.trim()) {
      return { content: [{ type: 'text', text: JSON.stringify({ changed_files: [], affected_symbols: [], affected_processes: [], risk_summary: { total_symbols: 0, total_processes: 0, risk_level: 'LOW' } }) }] };
    }

    // Parse unified diff into { file, addedRanges: [start, end][] }
    interface FileDiff { file: string; ranges: Array<[number, number]> }
    const fileDiffs: FileDiff[] = [];
    let current: FileDiff | null = null;

    for (const line of diffOutput.split('\n')) {
      if (line.startsWith('+++ b/')) {
        const file = line.slice(6);
        current = { file: path.join(repoPath, file), ranges: [] };
        fileDiffs.push(current);
      } else if (line.startsWith('@@ ') && current) {
        // @@ -old +new,count @@ or @@ -old +new @@
        const m = line.match(/@@ [^+]*\+(\d+)(?:,(\d+))? @@/);
        if (m) {
          const start = parseInt(m[1], 10);
          const count = m[2] !== undefined ? parseInt(m[2], 10) : 1;
          if (count > 0) current.ranges.push([start, start + count - 1]);
        }
      }
    }

    // Query brain store for affected symbols
    const rid = makeRepoId(source);
    const affectedSymbols: Array<{ file: string; fn: string; line: number; change_type: string }> = [];
    const affectedFileSet = new Set<string>();

    for (const fd of fileDiffs) {
      const relFile = path.relative(repoPath, fd.file);
      affectedFileSet.add(relFile);
      if (fd.ranges.length === 0) continue;

      for (const [rangeStart, rangeEnd] of fd.ranges) {
        const fns = brainStore.getFnsInRange(rid, relFile, rangeStart, rangeEnd);
        for (const fn of fns) {
          affectedSymbols.push({ file: relFile, fn: fn.name, line: fn.line, change_type: 'modified' });
        }
      }
    }

    // Query affected processes
    const affectedProcesses = brainStore.getProcessesForFiles(rid, [...affectedFileSet]);

    // Risk assessment
    const totalSymbols = affectedSymbols.length;
    const totalProcesses = affectedProcesses.length;
    const entryPointChanged = affectedSymbols.some(s => ['main', 'index', 'app', 'server', 'cli', 'run'].includes(s.fn));
    const risk_level = entryPointChanged ? 'CRITICAL' :
      totalSymbols >= 10 || totalProcesses >= 3 ? 'HIGH' :
      totalSymbols >= 4 || totalProcesses >= 1 ? 'MEDIUM' : 'LOW';

    const out = {
      changed_files: [...affectedFileSet],
      affected_symbols: affectedSymbols.slice(0, 50),
      affected_processes: affectedProcesses.slice(0, 20),
      risk_summary: { total_symbols: totalSymbols, total_processes: totalProcesses, risk_level },
    };
    return { content: [{ type: 'text', text: JSON.stringify(out) }], structuredContent: out };
  }
);
```

```typescript
getFnsInRange(repoId: string, filePath: string, startLine: number, endLine: number): FnRecord[] {
  return this.db.prepare(
    `SELECT * FROM functions WHERE repo_id = ? AND file_path = ? AND line >= ? AND line <= ?`
  ).all(repoId, repoId + ':' + filePath, startLine, endLine) as FnRecord[];
}

getProcessesForFiles(repoId: string, filePaths: string[]): Array<{ process_name: string; entry_file: string; depth: number }> {
  if (filePaths.length === 0) return [];
  const placeholders = filePaths.map(() => '?').join(',');
  const fileIds = filePaths.map(f => repoId + ':' + f);
  return this.db.prepare(
    `SELECT DISTINCT process_name, file_path as entry_file, MIN(depth) as depth FROM processes WHERE repo_id = ? AND file_path IN (${placeholders}) GROUP BY process_name ORDER BY depth`
  ).all(repoId, ...fileIds) as Array<{ process_name: string; entry_file: string; depth: number }>;
}
```
