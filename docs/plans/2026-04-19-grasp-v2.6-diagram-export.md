// mcp/src/diagram-gen.ts
import type { AnalysisResult } from './types.js';

function sanitizeId(path: string): string {
  return path.replace(/[^a-zA-Z0-9_]/g, '_').replace(/^_+/, '');
}

function sanitizeLabel(path: string): string {
  return path.split('/').pop() ?? path;
}

export function generateMermaid(data: AnalysisResult, maxNodes: number): string {
  // Group files by layer
  const layerMap = new Map<string, typeof data.files>();
  for (const file of data.files) {
    const layer = file.layer ?? 'other';
    if (!layerMap.has(layer)) layerMap.set(layer, []);
    layerMap.get(layer)!.push(file);
  }

  // Build a set of high-importance files (by fan-in)
  const fanInMap = new Map<string, number>();
  for (const conn of data.connections) {
    fanInMap.set(conn.to, (fanInMap.get(conn.to) ?? 0) + 1);
  }

  // Pick top maxNodes files by fan-in
  const allFiles = [...data.files].sort((a, b) => (fanInMap.get(b.path) ?? 0) - (fanInMap.get(a.path) ?? 0));
  const includedPaths = new Set(allFiles.slice(0, maxNodes).map(f => f.path));

  // Collect security/issue file set for styling
  const securityFiles = new Set(data.security?.map((s: any) => s.file).filter(Boolean) ?? []);
  const circularFiles = new Set(
    data.patterns?.filter((p: any) => p.type === 'circular').flatMap((p: any) => p.files ?? []) ?? []
  );

  const lines: string[] = ['graph LR'];

  // Subgraphs per layer
  for (const [layer, files] of layerMap.entries()) {
    const layerFiles = files.filter(f => includedPaths.has(f.path));
    if (layerFiles.length === 0) continue;

    lines.push(`  subgraph ${layer}`);
    for (const file of layerFiles) {
      const id = sanitizeId(file.path);
      const label = sanitizeLabel(file.path);
      if (securityFiles.has(file.path)) {
        lines.push(`    ${id}["🔴 ${label}"]`);
      } else if (circularFiles.has(file.path)) {
        lines.push(`    ${id}["🟡 ${label}"]`);
      } else {
        lines.push(`    ${id}["${label}"]`);
      }
    }
    lines.push('  end');
  }

  lines.push('');

  // Edges (only between included files)
  for (const conn of data.connections) {
    if (!includedPaths.has(conn.from) || !includedPaths.has(conn.to)) continue;
    lines.push(`  ${sanitizeId(conn.from)} --> ${sanitizeId(conn.to)}`);
  }

  return lines.join('\n');
}

export function generateC4Context(data: AnalysisResult): string {
  const systemName = (data.source.split('/').pop() ?? 'System').replace(/[^a-zA-Z0-9 ]/g, ' ');
  const layers = [...new Set(data.files.map(f => f.layer ?? 'core').filter(Boolean))];

  const lines: string[] = [
    'C4Context',
    `  title System Context — ${systemName}`,
    '',
    `  System(${sanitizeId(systemName)}, "${systemName}", "Analysed codebase with ${data.files.length} files across ${layers.length} layers")`,
  ];

  // Detect external dependencies from connections that go outside known files
  const knownPaths = new Set(data.files.map(f => f.path));
  const externalDeps = new Set<string>();
  for (const conn of data.connections) {
    if (!knownPaths.has(conn.to) && conn.to.includes('/')) {
      const pkg = conn.to.split('/')[0];
      if (pkg && !pkg.startsWith('src') && !pkg.startsWith('.')) externalDeps.add(pkg);
    }
  }

  for (const dep of [...externalDeps].slice(0, 8)) {
    lines.push(`  System_Ext(${sanitizeId(dep)}, "${dep}", "External dependency")`);
  }

  lines.push('');
  for (const dep of [...externalDeps].slice(0, 8)) {
    lines.push(`  Rel(${sanitizeId(systemName)}, ${sanitizeId(dep)}, "Uses")`);
  }

  return lines.join('\n');
}

export function generateC4Container(data: AnalysisResult): string {
  const systemName = (data.source.split('/').pop() ?? 'System').replace(/[^a-zA-Z0-9 ]/g, ' ');
  const layers = [...new Set(data.files.map(f => f.layer ?? 'core').filter(Boolean))];

  const lines: string[] = [
    'C4Container',
    `  title Container Diagram — ${systemName}`,
    '',
    `  System_Boundary(sys, "${systemName}") {`,
  ];

  const layerTech: Record<string, string> = {
    ui: 'TypeScript, React', services: 'TypeScript, Node.js', db: 'Database Layer',
    api: 'REST API', config: 'Configuration', utils: 'Utilities', models: 'Data Models',
  };

  for (const layer of layers) {
    const count = data.files.filter(f => (f.layer ?? 'core') === layer).length;
    const tech = layerTech[layer] ?? 'TypeScript';
    lines.push(`    Container(${sanitizeId(layer)}, "${layer}", "${tech}", "${count} files")`);
  }

  lines.push('  }');
  lines.push('');

  // Add inter-layer connections
  const layerConnMap = new Map<string, Set<string>>();
  for (const conn of data.connections) {
    const fromFile = data.files.find(f => f.path === conn.from);
    const toFile = data.files.find(f => f.path === conn.to);
    if (!fromFile || !toFile) continue;
    const fromLayer = fromFile.layer ?? 'core';
    const toLayer = toFile.layer ?? 'core';
    if (fromLayer === toLayer) continue;
    if (!layerConnMap.has(fromLayer)) layerConnMap.set(fromLayer, new Set());
    layerConnMap.get(fromLayer)!.add(toLayer);
  }

  for (const [from, tos] of layerConnMap.entries()) {
    for (const to of tos) {
      lines.push(`  Rel(${sanitizeId(from)}, ${sanitizeId(to)}, "imports")`);
    }
  }

  return lines.join('\n');
}

export function generateC4Component(data: AnalysisResult, targetLayer: string, maxNodes: number): string {
  const files = data.files.filter(f => (f.layer ?? 'core') === targetLayer).slice(0, maxNodes);
  const systemName = data.source.split('/').pop() ?? 'System';

  const lines: string[] = [
    'C4Component',
    `  title Component Diagram — ${targetLayer} layer of ${systemName}`,
    '',
    `  Container_Boundary(layer, "${targetLayer}") {`,
  ];

  for (const file of files) {
    const label = sanitizeLabel(file.path);
    const id = sanitizeId(file.path);
    const fnCount = file.functions?.length ?? 0;
    lines.push(`    Component(${id}, "${label}", "TypeScript", "${fnCount} functions")`);
  }

  lines.push('  }');
  lines.push('');

  const filePaths = new Set(files.map(f => f.path));
  for (const conn of data.connections) {
    if (!filePaths.has(conn.from) || !filePaths.has(conn.to)) continue;
    lines.push(`  Rel(${sanitizeId(conn.from)}, ${sanitizeId(conn.to)}, "imports")`);
  }

  return lines.join('\n');
}

server.registerTool('grasp_diagram', {
  title: 'Architecture Diagram Generator',
  description: `Generates architecture diagrams from analysis data. Output can be pasted directly into GitHub wikis, Notion, Confluence, or rendered by any Mermaid-compatible tool.

Formats:
  mermaid (default) — Mermaid flowchart grouped by architectural layer. 🔴 = security issue, 🟡 = circular dep.
  c4-context — C4 Context diagram: system box + detected external dependencies
  c4-container — C4 Container diagram: each layer as a container with inter-layer arrows
  c4-component — C4 Component diagram: individual files within a specific layer

Parameters:
  session_id: string — active analysis session
  format: "mermaid" | "c4-context" | "c4-container" | "c4-component" (default "mermaid")
  layer: string (only for c4-component) — layer to diagram
  max_nodes: number (default 50) — prune to top N files by fan-in when over limit

Returns:
  { diagram: string, format: string, nodeCount: number, tip: string }`,
  inputSchema: z.object({
    session_id: z.string(),
    format: z.enum(['mermaid', 'c4-context', 'c4-container', 'c4-component']).optional(),
    layer: z.string().optional(),
    max_nodes: z.number().int().min(5).max(200).optional(),
  }).strict(),
  annotations: { readOnlyHint: true, destructiveHint: false, idempotentHint: true, openWorldHint: false },
}, async ({ session_id, format = 'mermaid', layer, max_nodes = 50 }) => {
  const data = await getSession(session_id);
  if (!data) return { content: [{ type: 'text', text: `Session "${session_id}" not found. Run grasp_analyze first.` }] };

  let diagram: string;
  const nodeCount = Math.min(data.files.length, max_nodes);

  switch (format) {
    case 'mermaid':
      diagram = generateMermaid(data, max_nodes);
      break;
    case 'c4-context':
      diagram = generateC4Context(data);
      break;
    case 'c4-container':
      diagram = generateC4Container(data);
      break;
    case 'c4-component':
      if (!layer) return { content: [{ type: 'text', text: 'c4-component format requires the "layer" parameter.' }] };
      diagram = generateC4Component(data, layer, max_nodes);
      break;
    default:
      diagram = generateMermaid(data, max_nodes);
  }

  const tips: Record<string, string> = {
    mermaid: 'Paste into any GitHub Markdown file, Notion, or Confluence to render. Use ```mermaid code block.',
    'c4-context': 'Render with https://structurizr.com/ or any C4 PlantUML tool.',
    'c4-container': 'Shows how architectural layers connect. Ideal for technical onboarding docs.',
    'c4-component': `Shows individual files in the "${layer}" layer. Good for layer-specific code reviews.`,
  };

  const output = {
    session_id,
    source: data.source,
    format,
    nodeCount,
    diagram,
    tip: tips[format] ?? 'Paste into a compatible renderer.',
  };

  return {
    content: [{ type: 'text', text: `\`\`\`${format === 'mermaid' ? 'mermaid' : 'text'}\n${diagram}\n\`\`\`\n\n${tips[format]}` }],
    structuredContent: output,
  };
});
```

```typescript
server.registerTool('grasp_pr_review', {
  title: 'Inline PR Review Comments',
  description: `Posts inline code review comments on a GitHub Pull Request at the exact lines with high-severity findings. Unlike grasp_pr_comment (one summary comment), this posts per-line annotations visible in the PR diff view — like Reviewdog or CodeClimate.

Only posts for: complexity > 15, circular dependency participant, or security issue. Requires write access to the repo.

Parameters:
  session_id: string — active analysis session (must have been run against this repo)
  repo: string — "owner/repo" format
  pr_number: number — PR number to annotate
  token: string — GitHub PAT with repo write access
  min_complexity: number (default 15) — minimum cyclomatic complexity to flag
  dry_run: boolean (default false) — if true, returns comments that would be posted without posting

Returns:
  { posted: number, skipped: number, comments: [{path, line, body}], dryRun: boolean }`,
  inputSchema: z.object({
    session_id: z.string(),
    repo: z.string(),
    pr_number: z.number().int().positive(),
    token: z.string(),
    min_complexity: z.number().int().min(1).optional(),
    dry_run: z.boolean().optional(),
  }).strict(),
  annotations: { readOnlyHint: false, destructiveHint: false, idempotentHint: false, openWorldHint: true },
}, async ({ session_id, repo, pr_number, token, min_complexity = 15, dry_run = false }) => {
  const data = await getSession(session_id);
  if (!data) return { content: [{ type: 'text', text: `Session "${session_id}" not found. Run grasp_analyze first.` }] };

  const [owner, repoName] = repo.split('/');
  if (!owner || !repoName) return { content: [{ type: 'text', text: 'Invalid repo format. Use "owner/repo".' }] };

  const { Octokit } = await import('@octokit/rest');
  const octokit = new Octokit({ auth: token });

  // Fetch PR files to get diff positions
  let prFiles: any[];
  try {
    const { data: files } = await octokit.pulls.listFiles({ owner, repo: repoName, pull_number: pr_number, per_page: 100 });
    prFiles = files;
  } catch (err: any) {
    return { content: [{ type: 'text', text: `Failed to fetch PR files: ${err.message}` }] };
  }

  // Build map of changed files and their diff positions
  const changedFilePaths = new Set(prFiles.map((f: any) => f.filename));

  // Build security issue map by file
  const securityByFile = new Map<string, any[]>();
  for (const sec of data.security ?? []) {
    if (!sec.file) continue;
    if (!securityByFile.has(sec.file)) securityByFile.set(sec.file, []);
    securityByFile.get(sec.file)!.push(sec);
  }

  // Build circular dep set
  const circularFiles = new Set(
    (data.patterns ?? []).filter((p: any) => p.type === 'circular').flatMap((p: any) => p.files ?? [])
  );

  // Build complexity map by file
  const complexityByFile = new Map<string, { fn: string; complexity: number }[]>();
  for (const file of data.files) {
    for (const fn of file.functions ?? []) {
      if ((fn.complexity ?? 0) >= min_complexity) {
        if (!complexityByFile.has(file.path)) complexityByFile.set(file.path, []);
        complexityByFile.get(file.path)!.push({ fn: fn.name, complexity: fn.complexity ?? 0 });
      }
    }
  }

  type ReviewComment = { path: string; position: number; body: string };
  const comments: ReviewComment[] = [];

  for (const prFile of prFiles) {
    const filePath = prFile.filename;
    const findings: string[] = [];

    const secFindings = securityByFile.get(filePath) ?? [];
    for (const sec of secFindings) {
      findings.push(`🔴 **Security:** ${sec.type ?? 'issue'} — ${sec.description ?? 'potential vulnerability'}`);
    }

    if (circularFiles.has(filePath)) {
      findings.push(`🟡 **Circular dependency:** This file participates in a dependency cycle. Adding new imports here increases risk.`);
    }

    const complexFindings = complexityByFile.get(filePath) ?? [];
    for (const { fn, complexity } of complexFindings) {
      findings.push(`🟠 **High complexity:** \`${fn}\` has cyclomatic complexity ${complexity} (threshold: ${min_complexity}). Consider breaking it down.`);
    }

    if (findings.length > 0 && prFile.patch) {
      // Post at position 1 (first hunk) — GitHub's diff position counting
      const position = 1;
      const body = `**Grasp Analysis Findings**\n\n${findings.join('\n\n')}\n\n*Generated by [grasp](https://github.com/ashfordeOU/grasp)*`;
      comments.push({ path: filePath, position, body });
    }
  }

  if (dry_run) {
    const output = { posted: 0, skipped: changedFilePaths.size - comments.length, comments, dryRun: true };
    return {
      content: [{ type: 'text', text: truncate(JSON.stringify(output, null, 2)) }],
      structuredContent: output,
    };
  }

  let posted = 0;
  if (comments.length > 0) {
    try {
      await octokit.pulls.createReview({
        owner,
        repo: repoName,
        pull_number: pr_number,
        event: 'COMMENT',
        body: `Grasp found ${comments.length} finding(s) in this PR. See inline comments below.`,
        comments: comments.map(c => ({ path: c.path, position: c.position, body: c.body })),
      });
      posted = comments.length;
    } catch (err: any) {
      return { content: [{ type: 'text', text: `Failed to post review: ${err.message}` }] };
    }
  }

  const output = {
    posted,
    skipped: changedFilePaths.size - comments.length,
    comments,
    dryRun: false,
    prUrl: `https://github.com/${repo}/pull/${pr_number}`,
  };

  return {
    content: [{ type: 'text', text: truncate(JSON.stringify(output, null, 2)) }],
    structuredContent: output,
  };
});
```
