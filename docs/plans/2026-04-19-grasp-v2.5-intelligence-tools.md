```typescript
// mcp/src/license-scanner.ts
import * as fs from 'fs';
import * as path from 'path';

export interface LicenseDep {
  name: string;
  version: string;
  license: string;
  licenseCategory: 'permissive' | 'copyleft' | 'unknown';
}

export interface LicenseViolation {
  name: string;
  license: string;
  licenseCategory: string;
}

export interface LicenseScanResult {
  dependencies: LicenseDep[];
  summary: {
    total: number;
    permissive: number;
    copyleft: number;
    unknown: number;
  };
  violations: LicenseViolation[];
}

const PERMISSIVE = new Set([
  'MIT', 'Apache-2.0', 'BSD-2-Clause', 'BSD-3-Clause', 'ISC',
  'CC0-1.0', 'Unlicense', '0BSD', 'BlueOak-1.0.0',
]);

const COPYLEFT = new Set([
  'GPL-2.0', 'GPL-2.0-only', 'GPL-2.0-or-later',
  'GPL-3.0', 'GPL-3.0-only', 'GPL-3.0-or-later',
  'LGPL-2.0', 'LGPL-2.1', 'LGPL-2.1-only', 'LGPL-2.1-or-later',
  'LGPL-3.0', 'LGPL-3.0-only', 'LGPL-3.0-or-later',
  'AGPL-3.0', 'AGPL-3.0-only', 'AGPL-3.0-or-later',
  'MPL-2.0', 'EUPL-1.2',
]);

function categorize(license: string): 'permissive' | 'copyleft' | 'unknown' {
  if (!license) return 'unknown';
  const norm = license.trim().toUpperCase();
  for (const p of PERMISSIVE) if (norm === p.toUpperCase()) return 'permissive';
  for (const c of COPYLEFT) if (norm === c.toUpperCase()) return 'copyleft';
  if (norm.includes('MIT') || norm.includes('BSD') || norm.includes('APACHE') || norm.includes('ISC')) return 'permissive';
  if (norm.includes('GPL') || norm.includes('AGPL')) return 'copyleft';
  return 'unknown';
}

async function scanNpmModules(projectRoot: string): Promise<LicenseDep[]> {
  const nodeModulesPath = path.join(projectRoot, 'node_modules');
  const deps: LicenseDep[] = [];
  if (!fs.existsSync(nodeModulesPath)) return deps;

  let entries: string[];
  try {
    entries = fs.readdirSync(nodeModulesPath);
  } catch {
    return deps;
  }

  for (const entry of entries) {
    if (entry.startsWith('.')) continue;
    const pkgPath = entry.startsWith('@')
      ? path.join(nodeModulesPath, entry)
      : path.join(nodeModulesPath, entry);

    if (entry.startsWith('@')) {
      try {
        const scoped = fs.readdirSync(pkgPath);
        for (const sub of scoped) {
          await readPackageJson(path.join(pkgPath, sub, 'package.json'), deps);
        }
      } catch { /* skip */ }
    } else {
      await readPackageJson(path.join(pkgPath, 'package.json'), deps);
    }
  }
  return deps;
}

async function readPackageJson(pkgJsonPath: string, deps: LicenseDep[]): Promise<void> {
  try {
    const raw = fs.readFileSync(pkgJsonPath, 'utf8');
    const pkg = JSON.parse(raw);
    const licenseRaw = typeof pkg.license === 'string'
      ? pkg.license
      : (pkg.license?.type ?? pkg.licenses?.[0]?.type ?? '');
    deps.push({
      name: pkg.name || path.basename(path.dirname(pkgJsonPath)),
      version: pkg.version || 'unknown',
      license: licenseRaw || 'UNLICENSED',
      licenseCategory: categorize(licenseRaw),
    });
  } catch { /* skip malformed packages */ }
}

export async function scanLicenses(
  projectRoot: string,
  allowedLicenses?: string[],
  flagCopyleft = true,
): Promise<LicenseScanResult> {
  const deps = await scanNpmModules(projectRoot);

  const summary = {
    total: deps.length,
    permissive: deps.filter(d => d.licenseCategory === 'permissive').length,
    copyleft: deps.filter(d => d.licenseCategory === 'copyleft').length,
    unknown: deps.filter(d => d.licenseCategory === 'unknown').length,
  };

  const violations: LicenseViolation[] = [];
  for (const dep of deps) {
    if (flagCopyleft && dep.licenseCategory === 'copyleft') {
      violations.push({ name: dep.name, license: dep.license, licenseCategory: dep.licenseCategory });
    } else if (allowedLicenses && allowedLicenses.length > 0 && !allowedLicenses.includes(dep.license)) {
      violations.push({ name: dep.name, license: dep.license, licenseCategory: dep.licenseCategory });
    }
  }

  return { dependencies: deps, summary, violations };
}
```

```typescript
  { dependencies: [{name, version, license, licenseCategory}], summary: {total, permissive, copyleft, unknown}, violations: [{name, license}] }`,

    session_id: z.string(),
```

```typescript
server.registerTool('grasp_onboard', {
  title: 'Onboarding Reading Path',
  description: `Produces an ordered reading path for a new engineer entering a specific area of the codebase. Given a topic/area query, returns 5–15 files sorted by "must understand first" order (topological by layer, highest fan-in first).

Parameters:
  session_id: string — active analysis session
  query: string — area to explore (e.g. "auth", "payments", "src/services/user.ts")
  limit: number (default 12) — max files to return

Returns:
  { path: [{file, layer, fanIn, whyFirst, connectsTo: string[]}], topic: string, totalMatched: number }`,
  inputSchema: z.object({
    session_id: z.string(),
    query: z.string(),
    limit: z.number().int().min(1).max(30).optional(),
  }).strict(),
  annotations: { readOnlyHint: true, destructiveHint: false, idempotentHint: true, openWorldHint: false },
}, async ({ session_id, query, limit = 12 }) => {
  const data = await getSession(session_id);
  if (!data) return { content: [{ type: 'text', text: `Session "${session_id}" not found. Run grasp_analyze first.` }] };

  const q = query.toLowerCase();

  // Find entry-point files matching the query (by path, layer name, or function name)
  const candidates = data.files.filter(f => {
    const fPath = f.path.toLowerCase();
    return fPath.includes(q) || (f.layer && f.layer.toLowerCase().includes(q));
  });

  if (candidates.length === 0) {
    return { content: [{ type: 'text', text: `No files matching "${query}" found in session.` }] };
  }

  // Build fan-in map from connections
  const fanInMap = new Map<string, number>();
  for (const conn of data.connections) {
    fanInMap.set(conn.to, (fanInMap.get(conn.to) ?? 0) + 1);
  }

  // Build dependents map for "connectsTo" context
  const dependentsMap = new Map<string, string[]>();
  for (const conn of data.connections) {
    if (!dependentsMap.has(conn.from)) dependentsMap.set(conn.from, []);
    dependentsMap.get(conn.from)!.push(conn.to);
  }

  // Layer priority order (lower = read first)
  const layerOrder: Record<string, number> = {
    config: 0, types: 1, utils: 2, models: 3, db: 4,
    services: 5, controllers: 6, api: 7, ui: 8, test: 9,
  };

  const scored = candidates.map(f => {
    const fanIn = fanInMap.get(f.path) ?? 0;
    const layerScore = layerOrder[f.layer?.toLowerCase() ?? ''] ?? 5;
    const score = layerScore * 10 - fanIn; // low layer + high fan-in = read first
    return { f, fanIn, layerScore, score };
  });

  scored.sort((a, b) => a.score - b.score);
  const top = scored.slice(0, limit);

  const readingPath = top.map(({ f, fanIn }) => ({
    file: f.path,
    layer: f.layer ?? 'unknown',
    fanIn,
    whyFirst: fanIn > 5
      ? `High fan-in (${fanIn} dependents) — many files depend on this`
      : f.layer === 'config' || f.layer === 'types'
        ? 'Foundation layer — establishes types and config others rely on'
        : 'Entry point for this area of the codebase',
    connectsTo: (dependentsMap.get(f.path) ?? []).slice(0, 5),
  }));

  const output = {
    session_id,
    topic: query,
    totalMatched: candidates.length,
    path: readingPath,
    tip: `Start at the top of this list and work downward. Each file builds on understanding from the ones above it.`,
  };

  return {
    content: [{ type: 'text', text: truncate(JSON.stringify(output, null, 2)) }],
    structuredContent: output,
  };
});
```

```typescript
server.registerTool('grasp_types', {
  title: 'Type Annotation Coverage',
  description: `Reports type annotation coverage per file — percentage of functions with full parameter and return type annotations. For TypeScript: counts typed parameters and explicit return types. For Python: counts PEP 484 annotations.

Parameters:
  session_id: string — active analysis session
  min_fan_in: number (default 1) — only report files with at least this many dependents
  limit: number (default 20) — max files to return (sorted by lowest coverage first)

Returns:
  { files: [{path, layer, fanIn, typedFunctions, totalFunctions, coveragePct}], averageCoverage: number, worstFiles: [{path, coveragePct, fanIn}] }`,
  inputSchema: z.object({
    session_id: z.string(),
    min_fan_in: z.number().int().min(0).optional(),
    limit: z.number().int().min(1).max(100).optional(),
  }).strict(),
  annotations: { readOnlyHint: true, destructiveHint: false, idempotentHint: true, openWorldHint: false },
}, async ({ session_id, min_fan_in = 1, limit = 20 }) => {
  const data = await getSession(session_id);
  if (!data) return { content: [{ type: 'text', text: `Session "${session_id}" not found. Run grasp_analyze first.` }] };

  // Build fan-in map
  const fanInMap = new Map<string, number>();
  for (const conn of data.connections) {
    fanInMap.set(conn.to, (fanInMap.get(conn.to) ?? 0) + 1);
  }

  type FileCoverage = {
    path: string;
    layer: string;
    fanIn: number;
    typedFunctions: number;
    totalFunctions: number;
    coveragePct: number;
  };

  const fileCoverages: FileCoverage[] = [];

  for (const file of data.files) {
    const ext = file.path.split('.').pop()?.toLowerCase() ?? '';
    const isTypescript = ext === 'ts' || ext === 'tsx';
    const isPython = ext === 'py';
    if (!isTypescript && !isPython) continue;

    const fanIn = fanInMap.get(file.path) ?? 0;
    if (fanIn < min_fan_in) continue;

    const totalFunctions = file.functions?.length ?? 0;
    if (totalFunctions === 0) continue;

    let typedFunctions = 0;
    if (file.functions) {
      for (const fn of file.functions) {
        // Count as typed if function has any type annotation info
        // We use function signature as a proxy: check for ':' in params or return type
        const hasTypes = fn.signature
          ? fn.signature.includes(':') || fn.signature.includes('=>')
          : false;
        if (hasTypes) typedFunctions++;
      }
    }

    const coveragePct = totalFunctions > 0 ? Math.round((typedFunctions / totalFunctions) * 100) : 0;

    fileCoverages.push({
      path: file.path,
      layer: file.layer ?? 'unknown',
      fanIn,
      typedFunctions,
      totalFunctions,
      coveragePct,
    });
  }

  // Sort by coverage ascending (worst first), then by fan-in descending
  fileCoverages.sort((a, b) => a.coveragePct - b.coveragePct || b.fanIn - a.fanIn);
  const worstFiles = fileCoverages.slice(0, limit);

  const averageCoverage = fileCoverages.length > 0
    ? Math.round(fileCoverages.reduce((s, f) => s + f.coveragePct, 0) / fileCoverages.length)
    : 0;

  const output = {
    session_id,
    source: data.source,
    averageCoverage,
    totalFilesAnalyzed: fileCoverages.length,
    files: worstFiles,
    advice: averageCoverage >= 80
      ? 'Good type coverage overall.'
      : `Average coverage is ${averageCoverage}%. Prioritize annotating high fan-in files first — they have the most impact.`,
  };

  return {
    content: [{ type: 'text', text: truncate(JSON.stringify(output, null, 2)) }],
    structuredContent: output,
  };
});
```

```typescript
// Inside grasp_suggest handler, after existing suggestions array is built,
// add effort/impact scoring and re-sort:

// Build fan-in map for effort scoring
const fanInMap = new Map<string, number>();
for (const conn of data.connections) {
  fanInMap.set(conn.to, (fanInMap.get(conn.to) ?? 0) + 1);
}

// Annotate each suggestion with effort/impact scores
const scoredSuggestions = suggestions.map((s: any) => {
  const file = data.files.find((f: any) => f.path === s.file);
  const fanIn = fanInMap.get(s.file ?? '') ?? 0;
  const fnCount = file?.functions?.length ?? 1;
  const churnScore = file?.churn ?? 0;

  // Effort: high fan-in + many functions + high churn = harder to change
  const effort = Math.min(100, Math.round(fanIn * 3 + fnCount * 0.5 + churnScore * 0.2));

  // Impact: severity weight × dependents count
  const severityWeight = s.severity === 'critical' ? 4 : s.severity === 'high' ? 3 : s.severity === 'medium' ? 2 : 1;
  const impact = Math.min(100, Math.round(severityWeight * 20 + fanIn * 2));

  const ratio = effort > 0 ? Math.round((impact / effort) * 100) / 100 : impact;

  return { ...s, effort, impact, effortToImpactRatio: ratio };
});

// Sort by effort-to-impact ratio descending (best bang-for-buck first)
scoredSuggestions.sort((a: any, b: any) => b.effortToImpactRatio - a.effortToImpactRatio);
```
