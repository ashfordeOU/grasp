```typescript
// mcp/src/parsers/go.ts
export interface GoImport {
  path: string;
  alias?: string;
  stdlib: boolean;
  internal: boolean;
  localPath?: string;  // relative path within module (for internal imports)
}

export interface GoParseResult {
  imports: GoImport[];
  packageName: string;
}

// Standard library packages don't have a dot in the first path segment
function isStdlib(importPath: string): boolean {
  const firstSegment = importPath.split('/')[0];
  return !firstSegment.includes('.');
}

export function parseGoImports(
  source: string,
  filePath: string,
  moduleName?: string,
): GoParseResult {
  const imports: GoImport[] = [];

  // Extract package name
  const pkgMatch = source.match(/^package\s+(\w+)/m);
  const packageName = pkgMatch?.[1] ?? '';

  // Single import: import "path" or import alias "path"
  const singleImportRe = /^import\s+(?:(\w+)\s+)?"([^"]+)"/gm;
  for (const match of source.matchAll(singleImportRe)) {
    const [, alias, importPath] = match;
    imports.push(buildImport(importPath, alias, moduleName));
  }

  // Block import: import ( ... )
  const blockRe = /import\s*\(([^)]+)\)/gs;
  for (const blockMatch of source.matchAll(blockRe)) {
    const block = blockMatch[1];
    // Each line in block: optional alias then "path"
    const lineRe = /^\s*(?:(\w+|_|\.)\s+)?"([^"]+)"/gm;
    for (const lineMatch of block.matchAll(lineRe)) {
      const [, alias, importPath] = lineMatch;
      imports.push(buildImport(importPath, alias === '_' || alias === '.' ? alias : alias, moduleName));
    }
  }

  return { imports, packageName };
}

function buildImport(importPath: string, alias: string | undefined, moduleName?: string): GoImport {
  const stdlib = isStdlib(importPath);
  let internal = false;
  let localPath: string | undefined;

  if (moduleName && importPath.startsWith(moduleName + '/')) {
    internal = true;
    localPath = importPath.slice(moduleName.length + 1);
  }

  return {
    path: importPath,
    ...(alias ? { alias } : {}),
    stdlib,
    internal,
    ...(localPath ? { localPath } : {}),
  };
}
```

```typescript
// mcp/src/parsers/rust.ts
export interface RustImport {
  path: string;       // full use path as written
  module: string;     // first meaningful segment (for dep graph)
  stdlib: boolean;
  internal: boolean;  // crate:: or super::
  reExport: boolean;  // pub use
}

export interface RustParseResult {
  imports: RustImport[];
  submodules: string[];   // from mod xxx; declarations
}

export function parseRustImports(source: string, _filePath: string): RustParseResult {
  const imports: RustImport[] = [];
  const submodules: string[] = [];

  // mod declarations: mod name;
  for (const match of source.matchAll(/^\s*(?:pub\s+)?mod\s+(\w+)\s*;/gm)) {
    submodules.push(match[1]);
  }

  // extern crate name;
  for (const match of source.matchAll(/^\s*extern\s+crate\s+(\w+)\s*;/gm)) {
    const name = match[1];
    imports.push({ path: name, module: name, stdlib: name === 'std', internal: false, reExport: false });
  }

  // use statements (pub use or use)
  const useRe = /^\s*(pub\s+)?use\s+([\w:{}*,\s]+?)\s*;/gm;
  for (const match of source.matchAll(useRe)) {
    const [, pubKw, usePath] = match;
    const reExport = !!pubKw;
    const trimmed = usePath.trim();

    // Parse the first segment for module identification
    const firstSeg = trimmed.split('::')[0].trim();
    const stdlib = firstSeg === 'std' || firstSeg === 'core' || firstSeg === 'alloc';
    const internal = firstSeg === 'crate' || firstSeg === 'super' || firstSeg === 'self';

    // Get actual module name from path
    let module = firstSeg;
    if (internal) {
      const parts = trimmed.split('::');
      module = parts[1]?.trim() ?? firstSeg;
    }

    imports.push({ path: trimmed, module, stdlib, internal, reExport });
  }

  return { imports, submodules };
}
```

```typescript
// mcp/src/parsers/java.ts
export interface JavaImport {
  fullPath: string;     // full import statement path
  package: string;      // package part (everything except class name)
  className: string;    // simple class or method name
  isStatic: boolean;
  wildcard: boolean;
  stdlib: boolean;
}

export interface JavaParseResult {
  imports: JavaImport[];
  packageName: string;
}

const STDLIB_PREFIXES = ['java.', 'javax.', 'jakarta.', 'com.sun.', 'sun.', 'jdk.'];

function isStdlib(fullPath: string): boolean {
  return STDLIB_PREFIXES.some(p => fullPath.startsWith(p));
}

export function parseJavaImports(source: string, _filePath: string): JavaParseResult {
  const imports: JavaImport[] = [];

  // Package declaration
  const pkgMatch = source.match(/^\s*package\s+([\w.]+)\s*;/m);
  const packageName = pkgMatch?.[1] ?? '';

  // import [static] path[.*];
  const importRe = /^\s*import\s+(static\s+)?([\w.]+?)(\.\*)?\s*;/gm;
  for (const match of source.matchAll(importRe)) {
    const [, staticKw, path, wildcardSuffix] = match;
    const isStatic = !!staticKw;
    const wildcard = !!wildcardSuffix;

    let pkg: string;
    let className: string;

    if (wildcard) {
      pkg = path;
      className = '*';
    } else {
      const lastDot = path.lastIndexOf('.');
      if (lastDot === -1) {
        pkg = '';
        className = path;
      } else {
        pkg = path.substring(0, lastDot);
        className = path.substring(lastDot + 1);
      }
    }

    imports.push({
      fullPath: path + (wildcard ? '.*' : ''),
      package: pkg,
      className,
      isStatic,
      wildcard,
      stdlib: isStdlib(path),
    });
  }

  return { imports, packageName };
}
```
