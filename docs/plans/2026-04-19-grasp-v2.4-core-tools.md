```typescript
// mcp/src/env-scanner.ts
export interface EnvVarResult {
  vars: EnvVar[];
  undocumented: string[];
  testOnly: string[];
  layerViolations: Array<{ name: string; file: string; layer: string }>;
}

export interface EnvVar {
  name: string;
  files: Array<{ path: string; line: number; layer: string }>;
  layers: string[];
  inEnvExample: boolean;
  testOnly: boolean;
}

const PATTERNS = [
  /process\.env\.([A-Z_][A-Z0-9_]+)/g,
  /os\.getenv\(['"]([A-Z_][A-Z0-9_]+)['"]\)/g,
  /os\.environ(?:\.get)?\[['"]([A-Z_][A-Z0-9_]+)['"]\]/g,
  /os\.environ\.get\(['"]([A-Z_][A-Z0-9_]+)['"]\)/g,
  /getenv\(['"]([A-Z_][A-Z0-9_]+)['"]\)/g,
  /config\.get\(['"]([A-Z_][A-Z0-9_]+)['"]\)/g,
  /ENV\[['"]([A-Z_][A-Z0-9_]+)['"]\]/g,
];

export function scanEnvVars(
  files: Array<{ path: string; content: string; layer: string; isTest: boolean }>,
  envExampleVars: string[],
): EnvVarResult {
  const varMap = new Map<string, EnvVar>();

  for (const file of files) {
    if (!file.content) continue;
    const lines = file.content.split('\n');
    lines.forEach((line, idx) => {
      for (const pattern of PATTERNS) {
        pattern.lastIndex = 0;
        let m: RegExpExecArray | null;
        while ((m = pattern.exec(line)) !== null) {
          const name = m[1];
          if (!varMap.has(name)) {
            varMap.set(name, { name, files: [], layers: [], inEnvExample: false, testOnly: false });
          }
          const entry = varMap.get(name)!;
          entry.files.push({ path: file.path, line: idx + 1, layer: file.layer });
          if (!entry.layers.includes(file.layer)) entry.layers.push(file.layer);
        }
      }
    });
  }

  const envExampleSet = new Set(envExampleVars);
  const allTestPaths = new Set(files.filter(f => f.isTest).map(f => f.path));

  const vars = Array.from(varMap.values()).map(v => ({
    ...v,
    inEnvExample: envExampleSet.has(v.name),
    testOnly: v.files.every(f => allTestPaths.has(f.path)),
  }));

  return {
    vars,
    undocumented: vars.filter(v => !v.inEnvExample).map(v => v.name),
    testOnly: vars.filter(v => v.testOnly).map(v => v.name),
    layerViolations: vars.flatMap(v =>
      v.files.filter(f => f.layer === 'ui' || f.layer === 'components').map(f => ({ name: v.name, file: f.path, layer: f.layer }))
    ),
  };
}

export function parseEnvExample(content: string): string[] {
  return content.split('\n')
    .map(l => l.trim())
    .filter(l => l && !l.startsWith('#'))
    .map(l => l.split('=')[0].trim())
    .filter(Boolean);
}
```

```typescript
// mcp/src/event-mapper.ts
export interface EventEntry {
  name: string;
  emitters: Array<{ file: string; line: number }>;
  subscribers: Array<{ file: string; line: number }>;
  orphaned: boolean; // emitted but never subscribed
  ghost: boolean;    // subscribed but never emitted
}

export interface EventMapResult {
  events: EventEntry[];
  totalEvents: number;
  orphanedCount: number;
  ghostCount: number;
}

const EMIT_PATTERNS = [
  /(?:emitter|eventBus|bus|ee|mitt|EventEmitter)\.emit\(['"`]([^'"`]+)['"`]/g,
  /dispatch(?:Event)?\((?:new \w+Event\(['"`]([^'"`]+)['"`]|{type:\s*['"`]([^'"`]+)['"`])/g,
  /\.publish\(['"`]([^'"`]+)['"`]/g,
  /\.trigger\(['"`]([^'"`]+)['"`]/g,
  /createAction\(['"`]([^'"`]+)['"`]/g,
  /store\.dispatch\(['"`]([^'"`]+)['"`]/g,
];

const SUBSCRIBE_PATTERNS = [
  /(?:emitter|eventBus|bus|ee|mitt|EventEmitter)\.(?:on|once|addListener)\(['"`]([^'"`]+)['"`]/g,
  /addEventListener\(['"`]([^'"`]+)['"`]/g,
  /\.subscribe\(['"`]([^'"`]+)['"`]/g,
  /\.on\(['"`]([^'"`]+)['"`]/g,
  /on\(['"`]([^'"`]+)['"`],/g,
];

function extractMatches(content: string, patterns: RegExp[]): Array<{ name: string; line: number }> {
  const matches: Array<{ name: string; line: number }> = [];
  const lines = content.split('\n');
  lines.forEach((line, idx) => {
    for (const pattern of patterns) {
      pattern.lastIndex = 0;
      let m: RegExpExecArray | null;
      while ((m = pattern.exec(line)) !== null) {
        const name = m[1] || m[2];
        if (name && name.length < 80) matches.push({ name, line: idx + 1 });
      }
    }
  });
  return matches;
}

export function mapEvents(
  files: Array<{ path: string; content: string; layer: string }>
): EventMapResult {
  const emitterMap = new Map<string, Array<{ file: string; line: number }>>();
  const subscriberMap = new Map<string, Array<{ file: string; line: number }>>();

  for (const file of files) {
    if (!file.content) continue;
    for (const { name, line } of extractMatches(file.content, EMIT_PATTERNS)) {
      if (!emitterMap.has(name)) emitterMap.set(name, []);
      emitterMap.get(name)!.push({ file: file.path, line });
    }
    for (const { name, line } of extractMatches(file.content, SUBSCRIBE_PATTERNS)) {
      if (!subscriberMap.has(name)) subscriberMap.set(name, []);
      subscriberMap.get(name)!.push({ file: file.path, line });
    }
  }

  const allNames = new Set([...emitterMap.keys(), ...subscriberMap.keys()]);
  const events: EventEntry[] = Array.from(allNames).map(name => {
    const emitters = emitterMap.get(name) ?? [];
    const subscribers = subscriberMap.get(name) ?? [];
    return { name, emitters, subscribers, orphaned: emitters.length > 0 && subscribers.length === 0, ghost: subscribers.length > 0 && emitters.length === 0 };
  });

  return {
    events: events.sort((a, b) => (b.emitters.length + b.subscribers.length) - (a.emitters.length + a.subscribers.length)),
    totalEvents: events.length,
    orphanedCount: events.filter(e => e.orphaned).length,
    ghostCount: events.filter(e => e.ghost).length,
  };
}
```

```typescript
// TOOL: grasp_stale
server.registerTool(
  'grasp_stale',
  {
    title: 'Stale Code Detector',
    description: `Find files that are active (imported by others) but abandoned — untouched for N months, no test counterpart, and high fan-in. The "this probably still works but nobody knows why" list. Different from dead code — these files ARE depended on.

Parameters:
  min_months — minimum months since last change (default: 3)
  min_fan_in — minimum number of files importing this one (default: 2)
  limit — max results (default: 20)`,
    inputSchema: z.object({
      session_id: z.string(),
      min_months: z.number().optional().default(3),
      min_fan_in: z.number().optional().default(2),
      limit: z.number().optional().default(20),
    }).strict(),
    annotations: { readOnlyHint: true, destructiveHint: false, idempotentHint: true, openWorldHint: false },
  },
  async ({ session_id, min_months, min_fan_in, limit }) => {
    const data = await getSession(session_id);
    if (!data) return { content: [{ type: 'text', text: `Session "${session_id}" not found.` }] };

    // Build fan-in map
    const fanInMap = new Map<string, number>();
    for (const c of data.connections) {
      fanInMap.set(c.target, (fanInMap.get(c.target) ?? 0) + 1);
    }

    // Build test coverage map (files that have a .test. or .spec. counterpart)
    const testedFiles = new Set<string>();
    for (const f of data.files) {
      if (f.path.includes('.test.') || f.path.includes('.spec.') || f.layer === 'test') {
        const base = f.path.replace(/\.(test|spec)\.(ts|js|tsx|jsx|py)$/, '');
        testedFiles.add(base);
      }
    }

    const maxFanIn = Math.max(...Array.from(fanInMap.values()), 1);
    const now = Date.now();

    interface StaleFile {
      path: string;
      layer: string;
      fan_in: number;
      churn: number;
      months_stale: number;
      has_test: boolean;
      staleness_score: number;
    }

    const stale: StaleFile[] = data.files
      .filter(f => f.isCode && f.churn === 0) // churn=0 means no recent commits in analyzed window
      .map(f => {
        const fanIn = fanInMap.get(f.path) ?? 0;
        const hasTest = testedFiles.has(f.path.replace(/\.(ts|js|tsx|jsx|py)$/, ''));
        // Use churn as proxy for staleness — churn=0 means untouched in git window
        // Score: time(0.4) + fanIn(0.35) + noTest(0.25)
        const monthsStale = min_months; // floor at min_months since we can't compute exact months without git log
        const score = Math.round(
          (Math.min(monthsStale, 24) / 24 * 0.4 + fanIn / maxFanIn * 0.35 + (hasTest ? 0 : 0.25)) * 100
        );
        return { path: f.path, layer: f.layer, fan_in: fanIn, churn: f.churn, months_stale: monthsStale, has_test: hasTest, staleness_score: score };
      })
      .filter(f => f.fan_in >= min_fan_in)
      .sort((a, b) => b.staleness_score - a.staleness_score)
      .slice(0, limit);

    const output = { stale_files: stale, count: stale.length, criteria: { min_months, min_fan_in }, note: 'Staleness = files with no recent commits (churn=0) + high fan-in + no test counterpart. High fan-in means changing these files carries risk.' };
    return { content: [{ type: 'text', text: truncate(JSON.stringify(output, null, 2)) }], structuredContent: output };
  }
);
```

```typescript
// TOOL: grasp_change_risk
server.registerTool(
  'grasp_change_risk',
  {
    title: 'Change Risk Score',
    description: `Given a list of changed files (PR diff), compute a composite risk score 0–100. Score = blast_radius(35%) + avg_complexity(25%) + churn_frequency(20%) + layer_violations(20%). Returns a single actionable number plus per-component breakdown.

≤33 = low risk · 34–66 = medium · 67+ = high risk`,
    inputSchema: z.object({
      session_id: z.string(),
      changed_files: z.array(z.string()).describe('List of changed file paths (from PR diff)'),
    }).strict(),
    annotations: { readOnlyHint: true, destructiveHint: false, idempotentHint: true, openWorldHint: false },
  },
  async ({ session_id, changed_files }) => {
    const data = await getSession(session_id);
    if (!data) return { content: [{ type: 'text', text: `Session "${session_id}" not found.` }] };

    const changedSet = new Set(changed_files);
    const metrics = data.files.filter(f => changedSet.has(f.path));
    if (metrics.length === 0) return { content: [{ type: 'text', text: 'None of the changed files found in session. Verify file paths match.' }] };

    // Fan-in map for blast radius
    const fanInMap = new Map<string, number>();
    for (const c of data.connections) fanInMap.set(c.target, (fanInMap.get(c.target) ?? 0) + 1);
    const maxFanIn = Math.max(...Array.from(fanInMap.values()), 1);
    const maxComplexity = Math.max(...data.files.map(f => f.complexity?.score ?? 0), 1);
    const maxChurn = Math.max(...data.files.map(f => f.churn), 1);

    // Layer violations introduced by these files
    const newViolations = data.layerViolations.filter(v => changedSet.has(v.from));
    const maxPossibleViolations = Math.max(data.layerViolations.length, 1);

    const blastRadius = metrics.reduce((sum, f) => sum + (fanInMap.get(f.path) ?? 0), 0) / (metrics.length * maxFanIn);
    const avgComplexity = metrics.reduce((sum, f) => sum + (f.complexity?.score ?? 0), 0) / (metrics.length * maxComplexity);
    const churnFreq = metrics.reduce((sum, f) => sum + f.churn, 0) / (metrics.length * maxChurn);
    const layerViolScore = Math.min(newViolations.length / maxPossibleViolations, 1);

    const score = Math.round((blastRadius * 0.35 + avgComplexity * 0.25 + churnFreq * 0.20 + layerViolScore * 0.20) * 100);
    const level = score <= 33 ? 'low' : score <= 66 ? 'medium' : 'high';

    const output = {
      risk_score: score,
      risk_level: level,
      changed_files_found: metrics.length,
      breakdown: {
        blast_radius: Math.round(blastRadius * 100),
        avg_complexity: Math.round(avgComplexity * 100),
        churn_frequency: Math.round(churnFreq * 100),
        layer_violations: Math.round(layerViolScore * 100),
      },
      layer_violations_introduced: newViolations.map(v => ({ from: v.from, to: v.to, fromLayer: v.fromLayer, toLayer: v.toLayer })),
      highest_risk_files: metrics.sort((a, b) => (fanInMap.get(b.path) ?? 0) - (fanInMap.get(a.path) ?? 0)).slice(0, 5).map(f => ({ path: f.path, fan_in: fanInMap.get(f.path) ?? 0, complexity: f.complexity?.score ?? 0, churn: f.churn })),
    };
    return { content: [{ type: 'text', text: JSON.stringify(output, null, 2) }], structuredContent: output };
  }
);
```

```typescript
// mcp/src/flag-tracker.ts
export interface FeatureFlag {
  name: string;
  provider: string;
  files: Array<{ path: string; line: number; layer: string }>;
  blastRadius: number; // fan-in count of files using this flag
}

export interface FlagTrackerResult {
  flags: FeatureFlag[];
  totalFlags: number;
  providers: string[];
}

const FLAG_PATTERNS: Array<{ regex: RegExp; provider: string }> = [
  { regex: /ldClient\.variation\(['"`]([^'"`]+)['"`]/g, provider: 'LaunchDarkly' },
  { regex: /ld\.variation\(['"`]([^'"`]+)['"`]/g, provider: 'LaunchDarkly' },
  { regex: /useFlags\(\)\.([a-zA-Z_][a-zA-Z0-9_-]+)/g, provider: 'LaunchDarkly' },
  { regex: /gb\.isOn\(['"`]([^'"`]+)['"`]/g, provider: 'GrowthBook' },
  { regex: /gb\.getFeatureValue\(['"`]([^'"`]+)['"`]/g, provider: 'GrowthBook' },
  { regex: /client\.getBooleanValue\(['"`]([^'"`]+)['"`]/g, provider: 'OpenFeature' },
  { regex: /flags\.get\(['"`]([^'"`]+)['"`]/g, provider: 'Custom' },
  { regex: /features\.enabled\(['"`]([^'"`]+)['"`]/g, provider: 'Custom' },
  { regex: /isFeatureEnabled\(['"`]([^'"`]+)['"`]/g, provider: 'Custom' },
  { regex: /process\.env\.(FEATURE_[A-Z0-9_]+)/g, provider: 'EnvVar' },
  { regex: /process\.env\.(FF_[A-Z0-9_]+)/g, provider: 'EnvVar' },
  { regex: /process\.env\.(ENABLE_[A-Z0-9_]+)/g, provider: 'EnvVar' },
];

export function scanFeatureFlags(
  files: Array<{ path: string; content: string; layer: string }>
): FlagTrackerResult {
  const flagMap = new Map<string, FeatureFlag>();

  for (const file of files) {
    if (!file.content) continue;
    const lines = file.content.split('\n');
    lines.forEach((line, idx) => {
      for (const { regex, provider } of FLAG_PATTERNS) {
        regex.lastIndex = 0;
        let m: RegExpExecArray | null;
        while ((m = regex.exec(line)) !== null) {
          const name = m[1];
          if (!flagMap.has(name)) flagMap.set(name, { name, provider, files: [], blastRadius: 0 });
          const entry = flagMap.get(name)!;
          entry.files.push({ path: file.path, line: idx + 1, layer: file.layer });
        }
      }
    });
  }

  const flags = Array.from(flagMap.values()).map(f => ({ ...f, blastRadius: new Set(f.files.map(x => x.path)).size }));
  const providers = [...new Set(flags.map(f => f.provider))];

  return { flags: flags.sort((a, b) => b.blastRadius - a.blastRadius), totalFlags: flags.length, providers };
}
```

```typescript
// mcp/src/perf-analyzer.ts
export interface PerfFinding {
  pattern: string;
  severity: 'critical' | 'warning';
  file: string;
  line: number;
  description: string;
  suggestion: string;
}

export interface PerfAnalysisResult {
  findings: PerfFinding[];
  totalFindings: number;
  criticalCount: number;
  warningCount: number;
  fileCount: number;
}

const LOOP_OPEN = /^\s*(?:for\s*[\s\S]*?\)|while\s*\(|\.forEach\s*\(|\.map\s*\(|\.filter\s*\()/;

const PERF_PATTERNS: Array<{ name: string; severity: 'critical'|'warning'; regex: RegExp; description: string; suggestion: string; requiresLoop?: boolean }> = [
  { name: 'orm-in-loop', severity: 'critical', regex: /await\s+\w+\.(?:find|findOne|findAll|findBy|query|select|where|save|create|update|delete)\s*\(/,  description: 'ORM/DB query inside a loop — causes N+1 queries', suggestion: 'Batch with findAll/whereIn outside the loop', requiresLoop: true },
  { name: 'serialize-in-loop', severity: 'warning', regex: /JSON\.(?:stringify|parse)\s*\(/, description: 'JSON serialization inside a loop — avoid on hot paths', suggestion: 'Serialize once outside the loop or use a streaming approach', requiresLoop: true },
  { name: 'sync-fs-in-hot-file', severity: 'warning', regex: /(?:fs\.readFileSync|fs\.writeFileSync|fs\.existsSync)\s*\(/, description: 'Synchronous file I/O — blocks the event loop', suggestion: 'Replace with fs.promises (async) equivalent' },
  { name: 'missing-await', severity: 'warning', regex: /(?<!await\s)(?<!\.\s*)\b(?:fetch|axios\.get|axios\.post|db\.|prisma\.|mongoose\.)\w+\s*\(/, description: 'Async call without await — unhandled promise', suggestion: 'Add await or handle the returned Promise explicitly' },
  { name: 'regex-in-loop', severity: 'warning', regex: /new RegExp\s*\(/, description: 'RegExp constructor inside a loop — recompiles every iteration', suggestion: 'Move RegExp instantiation outside the loop as a const', requiresLoop: true },
];

export function analyzePerf(
  files: Array<{ path: string; content: string; layer: string }>
): PerfAnalysisResult {
  const findings: PerfFinding[] = [];

  for (const file of files) {
    if (!file.content) continue;
    const lines = file.content.split('\n');
    let inLoop = false;
    let loopDepth = 0;

    lines.forEach((line, idx) => {
      if (LOOP_OPEN.test(line)) { inLoop = true; loopDepth++; }
      const openBraces = (line.match(/\{/g) || []).length;
      const closeBraces = (line.match(/\}/g) || []).length;
      loopDepth = Math.max(0, loopDepth + openBraces - closeBraces);
      if (loopDepth === 0) inLoop = false;

      for (const p of PERF_PATTERNS) {
        if (p.requiresLoop && !inLoop) continue;
        if (p.regex.test(line)) {
          findings.push({ pattern: p.name, severity: p.severity, file: file.path, line: idx + 1, description: p.description, suggestion: p.suggestion });
        }
      }
    });
  }

  return {
    findings: findings.sort((a, b) => (a.severity === 'critical' ? -1 : 1) - (b.severity === 'critical' ? -1 : 1)),
    totalFindings: findings.length,
    criticalCount: findings.filter(f => f.severity === 'critical').length,
    warningCount: findings.filter(f => f.severity === 'warning').length,
    fileCount: new Set(findings.map(f => f.file)).size,
  };
}
```
