// Add to cli.ts — export for testability
export function renderProgressBar(done: number, total: number, file: string, width = 30): string {
  const pct = total > 0 ? done / total : 0;
  const filled = Math.round(pct * width);
  const empty = width - filled;
  const bar = '█'.repeat(filled) + '░'.repeat(empty);
  const filename = file.split('/').slice(-2).join('/'); // last 2 path segments
  return `Analyzing... [${bar}] ${done}/${total} ${filename}`;
}

// In the analyze command handler, after getting file count, add progress callback:
const progressCallback = (done: number, total: number, file: string) => {
  process.stderr.write('\r' + renderProgressBar(done, total, file));
  if (done === total) process.stderr.write('\n');
};

// Pass progressCallback as third argument to analyzeSource(...)

3. Add `debounce` export and `--watch` flag to `cli.ts`:
```typescript
// Add to cli.ts — export for testability
export function debounce<T extends (...args: any[]) => void>(fn: T, ms: number): T {
  let timer: ReturnType<typeof setTimeout> | undefined;
  return function (...args: Parameters<T>) {
    if (timer) clearTimeout(timer);
    timer = setTimeout(() => fn(...args), ms);
  } as T;
}

// Add --watch option to the analyze command:
.option('-w, --watch', 'Re-analyze on file changes (500ms debounce)')

// In analyze command handler, after running initial analysis, add:
if (options.watch) {
  console.error(`\nWatching ${resolvedPath} for changes. Press Ctrl+C to exit.\n`);

  let previousSession: any = null;

  const reanalyze = debounce(async () => {
    console.error('Change detected — re-analyzing...');
    try {
      const newResult = await analyzeSource(
        { type: 'local', path: resolvedPath },
        undefined,
        (done, total, file) => process.stderr.write('\r' + renderProgressBar(done, total, file)),
      );
      process.stderr.write('\n');

      if (previousSession) {
        // Show delta summary
        const oldScore = previousSession.summary.healthScore;
        const newScore = newResult.summary.healthScore;
        const delta = newScore - oldScore;
        const sign = delta >= 0 ? '+' : '';
        console.log(`Health: ${newScore} (${sign}${delta}) | Issues: ${newResult.summary.issueCount} | Files: ${newResult.summary.fileCount}`);
      } else {
        console.log(`Health: ${newResult.summary.healthScore} | Issues: ${newResult.summary.issueCount} | Files: ${newResult.summary.fileCount}`);
      }

      previousSession = newResult;
    } catch (err: any) {
      console.error('Re-analysis failed:', err.message);
    }
  }, 500);

  const watcher = fs.watch(resolvedPath, { recursive: true }, (event, filename) => {
    if (!filename) return;
    const ext = filename.split('.').pop()?.toLowerCase();
    const codeExts = new Set(['ts', 'tsx', 'js', 'jsx', 'py', 'go', 'rs', 'java', 'rb', 'php']);
    if (ext && codeExts.has(ext)) reanalyze();
  });

  // Keep process alive until Ctrl+C
  process.on('SIGINT', () => {
    watcher.close();
    console.error('\nWatch mode stopped.');
    process.exit(0);
  });

  // Block indefinitely
  await new Promise(() => {});
}
```

const GRADE_COLORS: Record<string, string> = {
  'A+': '#4c1', A: '#4c1', B: '#97ca00',
  C: '#dfb317', D: '#e05', F: '#e05',
};

export function generateBadgeSvg(grade: string, score: number): string {
  const color = GRADE_COLORS[grade] ?? '#9f9f9f';
  const label = 'grasp';
  const value = `${grade} (${score})`;

  // Shields.io-style flat badge SVG
  return `<svg xmlns="http://www.w3.org/2000/svg" xmlns:xlink="http://www.w3.org/1999/xlink" width="120" height="20">
  <linearGradient id="s" x2="0" y2="100%">

    <stop offset="1" stop-opacity=".1"/>
  </linearGradient>
  <clipPath id="r"><rect width="120" height="20" rx="3" fill="#fff"/></clipPath>
  <g clip-path="url(#r)">
    <rect width="45" height="20" fill="#555"/>
    <rect x="45" width="75" height="20" fill="${color}"/>
    <rect width="120" height="20" fill="url(#s)"/>
  </g>
  <g fill="#fff" text-anchor="middle" font-family="DejaVu Sans,Verdana,Geneva,sans-serif" font-size="110">

    <text x="235" y="140" transform="scale(.1)" textLength="350">${label}</text>

    <text x="825" y="140" transform="scale(.1)" textLength="650">${value}</text>

}
